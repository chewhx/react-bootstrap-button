## Overview

Bootstrap styled button with loading state

export const Loading = () => {
	const [loading, setLoading] = React.useState(false);
	const toggle = () => {
		setLoading(true);
		setTimeout(() => {
			setLoading(false);
		}, 4000);
	};
	return (
		<Button isLoading={loading} onClick={toggle} leftIcon={<FaFireAlt />}>
			Click to action
		</Button>
	);
};
